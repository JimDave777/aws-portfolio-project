document.getElementById("contactForm").addEventListener("submit", async function(e) {
    e.preventDefault();

    const responseDiv = document.getElementById("responseMessage");
    responseDiv.innerHTML = "Sending...";
    responseDiv.style.color = "white";

    const data = {
        name: document.getElementById("name").value,
        email: document.getElementById("email").value,
        message: document.getElementById("message").value
    };

    try {
        const response = await fetch("https://b835kr1xa6.execute-api.us-east-1.amazonaws.com/prod/contact", {
            method: "POST",
            headers: {
                "Content-Type": "application/json"
            },
            body: JSON.stringify(data)
        });

        const result = await response.json();

        if (response.ok) {
            responseDiv.innerHTML = "Message sent successfully!";
            responseDiv.style.color = "lightgreen";
            document.getElementById("contactForm").reset();
        } else {
            responseDiv.innerHTML = "Something went wrong.";
            responseDiv.style.color = "red";
        }

    } catch (error) {
        responseDiv.innerHTML = "Error connecting to server.";
        responseDiv.style.color = "red";
    }
});
</script>
